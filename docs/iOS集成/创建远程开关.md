# 创建远程开关

在开发阶段，工程师通过此方法在网站上创建远程开关

```swift
/**
 - parameter showName: 显示在网页上的名字，可以在网页修改
 - parameter varName: 开关的唯一值， 不可以和之前创建的重复
 - parameter value: 开关状态，可以在网页修改
*/
DYMobileSDK.createGlobalSwitch(globalSwitch: GlobalSwitch(showName: "showName", varName: "varName", value: true)) { results, error in
    if error == nil {
        if results?.errmsg == nil {
            // 创建成功
        }
    }
}
```

```objective-c
/**
 - parameter showName: 显示在网页上的名字，可以在网页修改
 - parameter varName: 开关的唯一值， 不可以和之前创建的重复
 - parameter value: 开关状态，可以在网页修改
*/
GlobalSwitch *globalSwitch = [[GlobalSwitch alloc] initWithShowName:@"ShowName" varName:@"varName" value:true];
[DYMobileSDK createGlobalSwitchWithGlobalSwitch:globalSwitch completion:^(SimpleStatusResult * data, NSError * error) {
    if (error == nil && data.errmsg == nil) {
        // 创建成功
    }
}];
```
